def generate_barnsley(A):

    df = pd.DataFrame([[np.array([[row[0], row[1]], [row[2], row[3]]]), 
                        np.array([[row[4]], [row[5]]]), 
                        row[6]] for row in A],
                     columns=["A", "B", "p"])

    n = 100000
    pts = np.empty((n, 2))
    pts[0] = [0, 0]

    choices = np.random.choice(len(A), n, p=df['p'])

    for row in range(1, n):
        i = choices[row]
        pts[row]=(np.matmul(df["A"][i], pts[row-1]).reshape((2, 1))+df["B"][i]).reshape((1, 2))

    plt.scatter(*zip(*pts), s=0.1, c='g')

Here is the classic Barnsley fern resembling the black spleenwort, Asplenium adiantum-nigrum.
